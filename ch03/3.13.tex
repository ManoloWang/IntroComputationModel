\problem{证明: 若在系统$\lambda\beta$中加入如下公理
$$\textrm{(A)}\qquad\lambda xy.x=\lambda xy.y,$$
则对任何的$M,N\in\Lambda$, $\lambda\beta+(A)\vdash M=N$.}
\begin{proof}
    对于所有的$M,N\in\Lambda$,
	\begin{align*}
		\lambda xy.x=\lambda xy.y&\Rightarrow (\lambda xy.x)MN=(\lambda xy.y)MN\\
		&\Rightarrow M=N
	\end{align*}
\end{proof}
